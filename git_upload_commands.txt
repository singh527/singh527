# Quick Git Upload Commands

# Navigate to your profile folder
cd d:\Sid\singh527

# Initialize git repository
git init

# Add all files
git add .

# Commit the files
git commit -m "Initial commit: Add GitHub profile README with cyberpunk theme"

# Add your GitHub repository as remote (replace with your actual repo URL)
git remote add origin https://github.com/singh527/singh527.git

# Push to GitHub
git branch -M main
git push -u origin main
